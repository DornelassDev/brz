Condicionais

03_condicionais.brz 
numero nota = 85

se (nota >= 90) {
    mostrar("Nota A")
} senao se (nota >= 80) {
    mostrar("Nota B")
} senao se (nota >= 70) {
    mostrar("Nota C")
} senao {
    mostrar("Reprovado")
}

// Ternario
texto status = (nota >= 60) ? "Aprovado" : "Reprovado"
mostrar(status)

// Escolha
texto dia = "segunda"
escolha (dia) {
    caso "segunda": { mostrar("Inicio da semana") }
    caso "sexta": { mostrar("Quase fim de semana") }
    padrao: { mostrar("Dia normal") }
}

Lacos de repeticao

04_lacos.brz 
// Para
para (numero i = 1; i <= 5; i++) {
    mostrar("Contagem: " + i)
}

// Para cada
lista texto frutas = ["maca", "banana", "laranja"]
para cada (fruta em frutas) {
    mostrar(fruta)
}

// Enquanto
numero n = 5
enquanto (n > 0) {
    mostrar(n)
    n -= 1
}

// Faca-enquanto
numero x = 0
faca {
    x += 1
} enquanto (x < 3)

// Pare e continue
para (numero i = 0; i < 10; i++) {
    se (i == 3) { continue }
    se (i == 7) { pare }
    mostrar(i)
}

Funcoes

06_funcoes.brz 
funcao numero somar(numero a, numero b) {
    retorne a + b
}

funcao vazio saudar(texto nome) {
    mostrar("Ola, " + nome + "!")
}

funcao numero fatorial(numero n) {
    se (n <= 1) { retorne 1 }
    retorne n * fatorial(n - 1)
}

funcao texto classificar(numero nota) {
    se (nota >= 90) { retorne "A" }
    se (nota >= 80) { retorne "B" }
    se (nota >= 70) { retorne "C" }
    retorne "F"
}

mostrar(somar(10, 20))
saudar("Carlos")
mostrar("5! = " + fatorial(5))
mostrar("Nota 85: " + classificar(85))

Classes e objetos

07_classes.brz 
classe Pessoa {
    texto nome
    numero idade

    construtor(texto nome, numero idade) {
        este.nome = nome
        este.idade = idade
    }

    funcao texto apresentar() {
        retorne "Sou " + este.nome + ", " + este.idade + " anos."
    }
}

classe ContaBancaria {
    texto titular
    decimal saldo

    construtor(texto titular, decimal saldoInicial) {
        este.titular = titular
        este.saldo = saldoInicial
    }

    funcao vazio depositar(decimal valor) {
        este.saldo += valor
    }

    funcao logico sacar(decimal valor) {
        se (valor > este.saldo) { retorne falso }
        este.saldo -= valor
        retorne verdadeiro
    }
}

Pessoa p = novo Pessoa("Joao", 30)
mostrar(p.apresentar())

ContaBancaria c = novo ContaBancaria("Ana", 1000.0)
c.depositar(500.0)
c.sacar(200.0)
mostrar("Saldo: " + c.saldo)

Tratamento de erros

10_erros.brz 
funcao decimal dividir(decimal a, decimal b) {
    se (b == 0) {
        lance("Divisao por zero!")
    }
    retorne a / b
}

tente {
    decimal r = dividir(10.0, 3.0)
    mostrar("10 / 3 = " + r)

    decimal r2 = dividir(10.0, 0.0)
    mostrar("Nunca chega aqui")
} pegue (erro) {
    mostrar("Erro: " + erro)
} finalmente {
    mostrar("Bloco finalmente executou.")
}

// Validacao
funcao vazio validarIdade(numero idade) {
    se (idade < 0) {
        lance("Idade negativa!")
    }
}

tente {
    validarIdade(-5)
} pegue (erro) {
    mostrar("Validacao: " + erro)
}
